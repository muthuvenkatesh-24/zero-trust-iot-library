44void ZtTlsHandshake::StartHandshake(Ptr<Node> client, Ptr<Node> server, uint32_t clientId, uint32_t serverId) {
45 NS_LOG_FUNCTION(this << client << server);
46
47 if (m_policyValidator && !m_policyValidator(clientId, "client")) {
48 Log("[ZT-HANDSHAKE] Client not authorized by policy");
49 return;
50 }
51
52 if (m_policyValidator && !m_policyValidator(serverId, "server")) {
53 Log("[ZT-HANDSHAKE] Server not authorized by policy");
54 return;
55 }
56
57 CryptoPP::AutoSeededRandomPool prng;
58 CryptoPP::byte key[CryptoPP::AES::DEFAULT_KEYLENGTH];
59 prng.GenerateBlock(key, sizeof(key));
60
61 std::string encoded;
62 CryptoPP::StringSource ss(key, sizeof(key), true,
63 new CryptoPP::HexEncoder(new CryptoPP::StringSink(encoded)));
64
65 m_sessionKeys[serverId] = encoded;
66 m_sessionKeys[clientId] = encoded;
67
68 Log("[ZT-HANDSHAKE] Session established between Client " + std::to_string(clientId) +
69 " and Server " + std::to_string(serverId) + " | Key: " + encoded);
70}
