65double
66ContextAttributeStore::CalculateDrift(uint32_t nodeId,
67 double wt,
68 double wc,
69 double wp)
70
71{
72 // Time deviation
73 double dt = std::abs(currentTime[nodeId] - baselineTime[nodeId]);
74 // Network deviation
75 double dc = std::abs(currentCongestion[nodeId] - baselineCongestion[nodeId]);
76 // Location deviation
77 double dp = std::abs((double)currentProximity[nodeId] -
78 (double)baselineProximity[nodeId]);
79
80 return wt * dt + wc * dc + wp * dp;
