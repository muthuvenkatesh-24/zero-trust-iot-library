#include "zt-identity-registry.h"
#include "zt-certificate.h"
#include "zt-policy-engine.h"
#include "zt-tls-handshake.h"
#include "zt-logger.h"
using namespace ns3;
// Register node identity
IdentityRegistry::GetInstance().RegisterNode(sensorNode, "sensor");
// Create Certificate Authority
CertificateAuthority ca;
// Issue certificate for node
std::string cert =
ca.SignIdentity(
1,
"sensor",
time(nullptr) + 86400
);
// Configure Zero Trust policy engine
Ptr<ZtPolicyEngine> policy =
CreateObject<ZtPolicyEngine>();
policy->AddAuthorized(
1,
"sensor"
);
// Establish TLS session
Ptr<ZtTlsHandshake> tls =
CreateObject<ZtTlsHandshake>();
tls->StartHandshake(
sensorNode,
gatewayNode,
1,
2
);
// Log successful authentication
ZtLogger::Log(
"INFO",
"Zero Trust session established"
);
