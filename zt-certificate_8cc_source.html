82bool ZtPolicyEngineWithCert::Authorize(uint32_t nodeId, const std::string& role, const std::string& certStr) {
83 using namespace ns3;
84
85 if (revoke.find(nodeId) != revoke.end()) {
86 NS_LOG_UNCOND("ZT-CERT: Node " << nodeId << " is revoked");
87 return false;
88 }
89
90 std::string content, sig;
91 size_t sigPos = certStr.find("|SIG:");
92 if (sigPos == std::string::npos) return false;
93 content = certStr.substr(0, sigPos);
94 sig = certStr.substr(sigPos + 5);
95
96 std::string decodedSig;
97 StringSource(sig, true, new Base64Decoder(new StringSink(decodedSig)));
98
99 RSASS<PSSR, SHA1>::Verifier verifier(caPublicKey);
100 bool valid = false;
101 StringSource(decodedSig + content, true,
102 new SignatureVerificationFilter(verifier,
103 new ArraySink((byte*)&valid, sizeof(valid)),
104 SignatureVerificationFilter::PUT_RESULT | SignatureVerificationFilter::SIGNATURE_AT_BEGIN));
105
106 if (!valid) {
107 NS_LOG_UNCOND("ZT-CERT: Signature invalid");
108 return false;
109 }
110
111 std::istringstream ss(content);
112 std::string token;
113 uint32_t idParsed = 0;
114 std::string roleParsed;
115 time_t expiry = 0;
116
117 while (std::getline(ss, token, '|')) {
118 if (token.find("ID:") == 0)
119 idParsed = std::stoul(token.substr(3));
120 else if (token.find("ROLE:") == 0)
121 roleParsed = token.substr(5);
122 else if (token.find("EXP:") == 0)
123 expiry = std::stol(token.substr(4));
124 }
125
126 if (idParsed != nodeId || roleParsed != role) {
127 NS_LOG_UNCOND("ZT-CERT: Identity mismatch");
128 return false;
129 }
130
131 time_t now = std::time(nullptr);
132 if (now > expiry) {
133 NS_LOG_UNCOND("ZT-CERT: Certificate expired");
134 return false;
135 }

137 return true;
138}
